74 --! @brief Architecture of signal generator
75 ARCHITECTURE my_signal_generator OF signal_generator IS
76 
77 CONSTANT UNLIMITED_STEPS : INTEGER := -1;
78 CONSTANT LEFT : STD_LOGIC := '0';
79 CONSTANT RIGHT : STD_LOGIC := NOT LEFT;
80 
81 TYPE MODE_STATE_TYPE IS (IDLE, CR, COS_1_4, COS_1_2, COS_1, COS_2);
82 TYPE PWM_STATE_TYPE IS (ONE, TWO, THREE, FOUR);
83 
84 SIGNAL mode_state : MODE_STATE_TYPE;
85 SIGNAL pwm_state : PWM_STATE_TYPE;
86 SIGNAL speed_wire : INTEGER RANGE 0 TO 400;
87 SIGNAL old_speed_wire : INTEGER RANGE 0 TO 400;
88 SIGNAL pwm_5ms_counter : INTEGER RANGE 0 TO 400;
89 SIGNAL steps_to_run : INTEGER;
90 SIGNAL steps_counter : INTEGER;
91 SIGNAL motor_en_wire : STD_LOGIC_VECTOR(1 DOWNTO 0);
92 SIGNAL motor_pwm_wire : STD_LOGIC_VECTOR(3 DOWNTO 0);
93 SIGNAL ir_wire : STD_LOGIC;
94 SIGNAL steps_output_wire : INTEGER;
95 
96 BEGIN
97  --- MODE's
98  mode_state_machine : PROCESS (reset_n, clock, mode, ir_wire, run)
99  BEGIN
100  IF(reset_n = '0') THEN
101  mode_state <= IDLE;
102  ELSIF(rising_edge(clock)) THEN
103  CASE mode_state IS
104  WHEN IDLE =>
105  motor_en_wire <= "00";
106  ir_wire <= '0';
107  IF mode(1 DOWNTO 0) = "01" THEN
108  mode_state <= CR;
109  ELSIF mode = "0010" THEN
110  mode_state <= COS_1_4;
111  ELSIF mode = "0110" THEN
112  mode_state <= COS_1_2;
113  ELSIF mode = "1010" THEN
114  mode_state <= COS_1;
115  ELSIF mode = "1110" THEN
116  mode_state <= COS_2;
117  END IF;
118  WHEN CR =>
119  IF(run = '1') THEN
120  motor_en_wire <= "11";
121  ELSE
122  motor_en_wire <= "00";
123  END IF;
124  IF(mode(1 DOWNTO 0) = "00") THEN -- user stopped the motor
125  mode_state <= IDLE;
126  ELSIF mode(1 DOWNTO 0) = "01" THEN
127  mode_state <= CR;
128  ELSIF mode = "0010" THEN
129  mode_state <= COS_1_4;
130  ELSIF mode = "0110" THEN
131  mode_state <= COS_1_2;
132  ELSIF mode = "1010" THEN
133  mode_state <= COS_1;
134  ELSIF mode = "1110" THEN
135  mode_state <= COS_2;
136  END IF;
137  WHEN OTHERS => -- all CoS states
138  IF(run = '1') THEN
139  motor_en_wire <= "11";
140  ELSE
141  motor_en_wire <= "00";
142  END IF;
143  IF(mode(1 DOWNTO 0) = "00") THEN -- user stopped the motor
144  mode_state <= IDLE;
145  ELSIF mode(1 DOWNTO 0) = "01" THEN
146  mode_state <= CR;
147  ELSIF mode = "0010" THEN
148  mode_state <= COS_1_4;
149  ELSIF mode = "0110" THEN
150  mode_state <= COS_1_2;
151  ELSIF mode = "1010" THEN
152  mode_state <= COS_1;
153  ELSIF mode = "1110" THEN
154  mode_state <= COS_2;
155  END IF;
156  IF(steps_counter = steps_to_run) THEN
157  ir_wire <= '1';
158  mode_state <= IDLE;
159  END IF;
160  END CASE;
161  END IF;
162  END PROCESS;
163 
164  mode_processing : PROCESS (mode_state)
165  BEGIN
166  CASE mode_state IS
167  WHEN IDLE =>
168  steps_to_run <= UNLIMITED_STEPS;
169  WHEN CR =>
170  steps_to_run <= UNLIMITED_STEPS;
171  WHEN COS_1_4 =>
172  steps_to_run <= 100 - 1;
173  WHEN COS_1_2 =>
174  steps_to_run <= 200 - 1;
175  WHEN COS_1 =>
176  steps_to_run <= 400 - 1;
177  WHEN COS_2 =>
178  steps_to_run <= 800 - 1;
179  WHEN others =>
180  steps_to_run <= 0;
181  END CASE;
182  END PROCESS;
183 
184  --- SPEED's
185  --- this block generates 9 combinational loops with latches
186  speed_converting : PROCESS (reset_n, clock, speed, speed_wire)
187  BEGIN
188  old_speed_wire <= speed_wire;
189  CASE speed IS
190  WHEN "000" =>
191  speed_wire <= 400 - 1;
192  WHEN "001" =>
193  speed_wire <= 200 - 1;
194  WHEN "010" =>
195  speed_wire <= 100 - 1;
196  WHEN "011" =>
197  speed_wire <= 50 - 1;
198  WHEN "100" =>
199  speed_wire <= 20 - 1;
200  WHEN "101" =>
201  speed_wire <= 10 - 1;
202  WHEN "110" =>
203  speed_wire <= 5 - 1;
204  WHEN "111" =>
205  speed_wire <= 2 - 1;
206  WHEN others =>
207  speed_wire <= 400 - 1;
208  END CASE;
209  END PROCESS;
210 
211  --- PWM's
212  pulse_generation : PROCESS (reset_n, clock, prescaler, speed_wire, old_speed_wire)
213  BEGIN
214  IF(reset_n = '0') THEN
215  pwm_5ms_counter <= speed_wire;
216  ELSIF(speed_wire /= old_speed_wire) THEN
217  pwm_5ms_counter <= speed_wire;
218  ELSIF(rising_edge(clock) and prescaler = '1') THEN
219  IF(pwm_5ms_counter = 0) THEN
220  pwm_5ms_counter <= speed_wire;
221  ELSE
222  pwm_5ms_counter <= pwm_5ms_counter - 1;
223  END IF;
224  END IF;
225  END PROCESS;
226 
227  pwm_state_machine : PROCESS (reset_n, clock, prescaler, pwm_5ms_counter, direction, pwm_state)
228  BEGIN
229  IF(reset_n = '0') THEN
230  pwm_state <= ONE;
231  ELSIF(rising_edge(prescaler) AND pwm_5ms_counter = 0) THEN
232  IF(direction = LEFT) THEN
233  CASE pwm_state IS
234  WHEN FOUR =>
235  pwm_state <= THREE;
236  WHEN THREE =>
237  pwm_state <= TWO;
238  WHEN TWO =>
239  pwm_state <= ONE;
240  WHEN ONE =>
241  pwm_state <= FOUR;
242  END CASE;
243  ELSE
244  CASE pwm_state IS
245  WHEN ONE =>
246  pwm_state <= TWO;
247  WHEN TWO =>
248  pwm_state <= THREE;
249  WHEN THREE =>
250  pwm_state <= FOUR;
251  WHEN FOUR =>
252  pwm_state <= ONE;
253  END CASE;
254  END IF;
255  END IF;
256  END PROCESS;
257 
258  pwm_output : PROCESS (pwm_state)
259  BEGIN
260  CASE pwm_state IS
261  WHEN ONE =>
262  motor_pwm_wire <= "1100";
263  WHEN TWO =>
264  motor_pwm_wire <= "0110";
265  WHEN THREE =>
266  motor_pwm_wire <= "0011";
267  WHEN FOUR =>
268  motor_pwm_wire <= "1001";
269  END CASE;
270  END PROCESS;
271 
272  --- STEPS's
273  steps_counting : PROCESS (reset_n, prescaler, pwm_5ms_counter, pwm_state, run)
274  BEGIN
275  IF(reset_n = '0') THEN
276  steps_counter <= 0;
277  steps_output_wire <= 0;
278  ELSIF(run = '0') then
279  steps_counter <= 0;
280  steps_output_wire <= 0;
281  ELSIF(rising_edge(prescaler) AND pwm_5ms_counter = 0) THEN
282  IF(direction = LEFT) THEN
283  steps_output_wire <= steps_output_wire - 1;
284  ELSE
285  steps_output_wire <= steps_output_wire + 1;
286  END IF;
287  steps_counter <= steps_counter + 1;
288  END IF;
289  END PROCESS;
290 
291  motor_en <= motor_en_wire;
292  motor_pwm <= motor_pwm_wire;
293  ir <= ir_wire;
294  steps <= conv_std_logic_vector(steps_output_wire, 32);
295 END my_signal_generator;
