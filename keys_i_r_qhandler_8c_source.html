29 void keysIRQhandler(void *context) {
30  uint32_t edgesCaptured = 0;
31  INT8U err;
32 
33  OSIntEnter();
34  edgesCaptured = PIO_KEY_GetEdgeCpt();
35 
36  // Send corresponding Event and clear bit in edgecapture reg
37  if (edgesCaptured & PIO_KEY_RS_IR0_MSK) {
38  debug("KEY_RS_0 pressed!\n");
39  OSFlagPost(userInputTaskFlagsGrp, KEY0_RS_EVENT, OS_FLAG_SET, &err);
40  if (OS_NO_ERR != err) {
41  error("KEY_ISR_FLAG_ERR: %i\n", err);
42  }
43  }
44  if (edgesCaptured & PIO_KEY_MINUS_IR2_MSK) {
45  debug("KEY_MINUS_2 pressed!\n");
46  OSFlagPost(userInputTaskFlagsGrp, KEY2_MINUS_EVENT, OS_FLAG_SET, &err);
47  if (OS_NO_ERR != err) {
48  error("KEY_ISR_FLAG_ERR: %i\n", err);
49  }
50  }
51  if (edgesCaptured & PIO_KEY_PLUS_IR3_MSK) {
52  debug("KEY_PLUS_3 pressed!\n");
53  OSFlagPost(userInputTaskFlagsGrp, KEY3_PLUS_EVENT, OS_FLAG_SET, &err);
54  if (OS_NO_ERR != err) {
55  error("KEY_ISR_FLAG_ERR: %i\n", err);
56  }
57  }
58  PIO_KEY_ClearEdgeCptBits(edgesCaptured);
59  OSIntExit();
60 }
