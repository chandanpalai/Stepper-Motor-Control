55 uint8_t outputTaskDataTx(OS_EVENT* mutex, outputTaskData_t data){
56  uint8_t err;
57 
58  if(NULL == mutex){
59  err = OS_ERR_INVALID_OPT;
60  }else
61  OSMutexPend(mutex, 0, &err);
62  if(OS_NO_ERR == err){
63  outputTaskData.ctrlReg = data.ctrlReg;

65  outputTaskData.stepsReg = data.stepsReg;
