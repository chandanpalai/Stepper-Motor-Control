37 --! @brief Architecture of Key-Detector
38 --! @details a key is de-bounced and a detect signal
39 --! is generated (1 clock period)
40 architecture mykeydetect of keyDetect is
41 
42  --! Statedefinition for key detector
43  TYPE STATE_TYPE IS (start, newpress, pressed);
44  --! State signal for key detector
45  SIGNAL STkey: STATE_TYPE;
46 
47 begin
48  --! @brief detects and de-bounces the press of a key
49  --! at key-input-line. Generate a press detect
50  --! signal (1 clock period))
51  process (clock, reset_n, key_input)
52  begin
53  if (reset_n = '0') THEN
54  STkey <= start;
55  elsif (rising_edge(clock)) THEN
56  case STkey IS
57  when start =>
58  if(key_input = '0') THEN
59  STkey <= newpress;
60  else
61  STkey <= STkey;
62  end if;
63  when newpress =>
64  STkey <= pressed;
65  when pressed =>
66  if(key_input = '1') THEN
67  STkey <= start;
68  else
69  STkey <= STkey;
70  end if;
71  when others => STkey <= start;
72  end case;
73  end if;
74  end process;
75 
76  key_detect <= '1' WHEN STkey = newpress ELSE
77  '0';
78 end mykeydetect;
