35 void UserInputTask(void *pdata) {
36  uint8_t err;

41  uint32_t speedReg = 0;
42  uint32_t stepsReg = 0;
43 
44  //variable to store position of switches
45  uint32_t switchesReg = 0;
46 
47  systemState_t systemState = {
48  .operationalStatus = FUNCTIONAL,
49  .activeUseCase = STOP
50  };
51 
52  //create mailboxes
53  outputTaskMailbox_t outputTaskMailboxContent = {
54  .ctrlReg = 0,
55  .speedReg = 0,
56  .stepsReg = 0,
57  .debugState = false
58  };
59 
60  void *msg;
61  void *msgQueue[16];
62 
63  //Show initial terminal msg
64  printf_term("Stepper Motor - System on a Chip 2014\n");
65  printf_term("Michael Riedel & Marc Kossmann\n");
66  printf_term("Version 0.1 - 28.10.2014\n");
67 
68  //init LC-Display and show initial screen
69  init_lcd();
70  printf_lcd(" SoC 2014 \n");
71  printf_lcd("Stepper-Control ");
72  fflush_lcd();
73 
74  // hardwareTest();
75 
76  switchesMsgQueue = OSQCreate(msgQueue, 16);
77  outputTaskMailbox = OSMboxCreate(NULL);
78  if ((OS_EVENT *)0 == switchesMsgQueue || (OS_EVENT *)0 == outputTaskMailbox) {
79  error("INPUT_TASK_MBOX_MSGQ_ERR: %i\n", err);
80  } else {
81  while (0) {
82  //check for new events
83  newFlag = OSFlagPend(userInputTaskFlagsGrp,
84  (KEY0_RS_EVENT | KEY2_MINUS_EVENT | KEY3_PLUS_EVENT |
85  MOTOR_STOP_EVENT),
86  OS_FLAG_WAIT_SET_ANY + OS_FLAG_CONSUME, 10, &err);
87  if (OS_NO_ERR == err) {
88  //check for key0 event
89  if (newFlag & KEY0_RS_EVENT) {
90  //toogle RS-Bit in ctrlReg
91  if (ctrlReg & CTRL_REG_RS_MSK) { //RS-Bit is 1
92  ctrlReg &= ~(CTRL_REG_RS_MSK);
93  } else { //RS-Bit is 0
94  ctrlReg |= CTRL_REG_RS_MSK;
95  }
96  }
97  //check for key2 event (decrease steps)
98  if (newFlag & KEY2_MINUS_EVENT) {
99  if (!ctrlReg & CTRL_REG_RS_MSK) //allow modification only when motor stopped
100  speedReg--;
101  }
102  //check for key3 event (increase steps)
103  if (newFlag & KEY3_PLUS_EVENT) {
104  if (!ctrlReg & CTRL_REG_RS_MSK) //allow modification only when motor stopped
105  speedReg++;
106  }
107  //check for motor stop event
108  if (newFlag & MOTOR_STOP_EVENT) {
109  ctrlReg &= ~(CTRL_REG_RS_MSK);
110  }
111  } else if (OS_TIMEOUT != err) {
112  error("INPUT_TASK_FLAG_ERR: %i\n", err);
113  }
114 
115  //check for new switches message
116  msg = OSQPend(switchesMsgQueue, 10, &err);
117  if (OS_NO_ERR == err) {
118  switchesReg = (uint32_t)msg;
119  } else if (OS_TIMEOUT != err) {
120  error("INPUT_TASK_MBOX_ERR: %i\n", err);
121  }
122  //evaluate switch positions
123  if (switchesReg & SWITCH_LR_MSK) {
124  ctrlReg |= CTRL_REG_LR_MSK;
125  } else {
126  ctrlReg &= ~(CTRL_REG_LR_MSK);
127  }
128  if (!ctrlReg & CTRL_REG_RS_MSK) { //allow modification only when motor stopped
129  ctrlReg &= ~(SWITCH_MODE_MSK); //clear mode bits
130  ctrlReg |= (switchesReg & SWITCH_MODE_MSK); //set mode bits
131  }
132 
133  if (switchesReg & SWITCH_DEBUG_MSK) {
134  systemState.operationalStatus = DEBUG;
135  err = OSFlagPost(heartbeatTaskFlagsGrp, DEBUG_ON_EVENT, OS_FLAG_SET, NULL);
136  if (OS_NO_ERR != err) {
137  error("INPUT_TASK_FLAG_ERR: %i\n", err);
138  }
139  }
140 
141  //change systemState when Run-Bit = 1
142  if (ctrlReg & CTRL_REG_RS_MSK) {
143  if (((ctrlReg & CTRL_REG_MODE_MSK) & MODE_STOP_CON_RUN_MSK) ==
144  MODE_STOP) {
145  systemState.activeUseCase = STOP;
146  } else if (((ctrlReg & CTRL_REG_MODE_MSK) & MODE_STOP_CON_RUN_MSK) ==
147  MODE_CON_RUN) {
148  systemState.activeUseCase = CONTINOUS;
149  } else if ((ctrlReg & CTRL_REG_MODE_MSK) == MODE_CH_OF_ST_1_4) {
150  systemState.activeUseCase = QUARTER_ROTATION;
151  } else if ((ctrlReg & CTRL_REG_MODE_MSK) == MODE_CH_OF_ST_1_2) {
152  systemState.activeUseCase = HALF_ROTATION;
153  } else if ((ctrlReg & CTRL_REG_MODE_MSK) == MODE_CH_OF_ST_1) {
154  systemState.activeUseCase = FULL_ROTATION;
155  } else if ((ctrlReg & CTRL_REG_MODE_MSK) == MODE_CH_OF_ST_2) {
156  systemState.activeUseCase = DOUBLE_ROTATION;
157  }
158  }
159 
160  //update content of mailbox to outputTask
161  outputTaskMailboxContent.ctrlReg = ctrlReg;
162  outputTaskMailboxContent.speedReg = speedReg;
163  outputTaskMailboxContent.stepsReg = stepsReg;
164  if (systemState.operationalStatus == DEBUG) {
165  outputTaskMailboxContent.debugState = true;
166  } else {
167  outputTaskMailboxContent.debugState = false;
168  }
169  err = OSMboxPost(outputTaskMailbox, &outputTaskMailboxContent);
170  if (OS_NO_ERR != err) {
171  error("INPUT_TASK_MBOX_ERR: %i\n", err);
172  }
173  }
174  }
175 }
