47 void UserInputTask(void *pdata) {
48  uint8_t err;
49  uint32_t modeBits;
50  OS_FLAGS newFlag;
51 
52  bool newInput = false;
53 
54  // used to check for stepsReg update
55  uint32_t oldStepsReg = 0;
56  // register copies to work with
57  uint8_t ctrlReg = 0;
58  uint8_t speedReg = 0;
59  uint32_t stepsReg = 0;
60 
61  // variable to store position of switches
62  uint32_t switchesReg = 0;
63 
64  systemState_t systemState = {
65  .operationalStatus = FUNCTIONAL,
66  .activeUseCase = STOP };
67 
68  // declare mailbox
69  outputTaskMailbox_t outputTaskMailboxContent = {
70  .systemState = systemState,
71  .ctrlReg = 0,
72  .speedReg = 0,
73  .stepsReg = 0
74  };
75 
76  // show initial terminal msg
77  printf_term("Stepper Motor - System on a Chip 2014\n");
78  printf_term("Michael Riedel & Marc Kossmann\n");
79  printf_term("Version %s - %s\n", VERSION, DATE);
80 
81  // init LC-Display and show initial screen
82  init_lcd();
83  setPos_lcd(1, 5);
84  printf_lcd("SoC 2014");
85  setPos_lcd(2, 1);
86  printf_lcd("Stepper-Control");
87  fflush_lcd();
88 
89  // create OS resources
90  registerMutex = OSMutexCreate(3, &err);
91  // Enable Interrupt
92  #ifdef INTERRUPT_ENABLE
93  ctrlRegBitSet(CTRL_REG_IE_MSK);
94  #endif
95  ctrlRegBitClr(CTRL_REG_IR_MSK);
96  if ((OS_EVENT *) 0 == registerMutex || OS_NO_ERR != err) {
97  error("REGISTER_MUTEX_CREATE_ERR: %i\n", err);
98  } else {
99  outputTaskMailbox = OSMboxCreate(&outputTaskMailboxContent);
100  if ((OS_EVENT *) 0 == outputTaskMailbox) {
101  error("INPUT_TASK_MBOX_MSGQ_ERR: %i\n", err);
102  } else {
103  while (1) {
104  // get register copies
105  ctrlReg = ctrlRegGet();
106  speedReg = speedRegGet();
107  stepsReg = stepsRegGet();
108  // check for new events
109  newFlag =
110  OSFlagPend(userInputTaskFlagsGrp,
111  (KEY0_RS_EVENT | KEY2_MINUS_EVENT | KEY3_PLUS_EVENT
112  | MOTOR_STOP_EVENT| SW_UPDATE_EVENT), OS_FLAG_WAIT_SET_ANY + OS_FLAG_CONSUME,
113  10, &err);
114  if (OS_NO_ERR == err) {
115  newInput = true;
116  // check for key0 event
117  if (newFlag & KEY0_RS_EVENT) {
118  // toogle RS-Bit in ctrlReg
119  if (ctrlReg & CTRL_REG_RS_MSK) { //RS-Bit is 1
120  ctrlReg &= ~(CTRL_REG_RS_MSK);
121  } else { //RS-Bit is 0
122  ctrlReg |= CTRL_REG_RS_MSK;
123  }
124  }
125  // check for key2 event (decrease steps)
126  if (newFlag & KEY2_MINUS_EVENT) {
127  // is motor stopped?
128  if (!(ctrlReg & CTRL_REG_RS_MSK)) {
129  if (0 != speedReg) {
130  speedReg--;
131  }
132  }
133  }
134  // check for key3 event (increase steps)
135  if (newFlag & KEY3_PLUS_EVENT) {
136  // is motor stopped?
137  if (!(ctrlReg & CTRL_REG_RS_MSK)) {
138  if (7 != speedReg) {
139  speedReg++;
140  }
141  }
142  }
143  // check for motor stop event
144  if (newFlag & MOTOR_STOP_EVENT) {
145  ctrlReg &= ~(CTRL_REG_RS_MSK);
146  }
147  } else if (OS_TIMEOUT != err) {
148  error("INPUT_TASK_FLAG_ERR: %i\n", err);
149  }
150  // check for switches event
151  if (newFlag & SW_UPDATE_EVENT) {
152  // get current sw events
153  switchesReg = PIO_SW_GetValues();
154  // evaluate switch positions
155  if (switchesReg & PIO_SW_LR_MSK) {
156  ctrlReg |= CTRL_REG_LR_MSK;
157  } else {
158  ctrlReg &= ~(CTRL_REG_LR_MSK);
159  }
160  // is motor stopped?
161  if (!(ctrlReg & CTRL_REG_RS_MSK)) {
162  // then clear mode bits ...
163  ctrlReg &= ~(CTRL_REG_MODE_MSK);
164  // ... and set mode bits
165  ctrlReg |= ((switchesReg & PIO_SW_MODE_MSK) << 1);
166  }
167  if (switchesReg & PIO_SW_DEBUG_MSK) {
168  systemState.operationalStatus = DEBUG;
169  OSFlagPost(heartbeatTaskFlagsGrp, DEBUG_ON_EVENT, OS_FLAG_SET,
170  &err);
171  if (OS_NO_ERR != err) {
172  error("INPUT_TASK_FLAG_ERR: %i\n", err);
173  }
174  } else {
175  if(systemState.operationalStatus == DEBUG){
176  systemState.operationalStatus = FUNCTIONAL;
177  OSFlagPost(heartbeatTaskFlagsGrp, DEBUG_OFF_EVENT, OS_FLAG_SET,
178  &err);
179  if (OS_NO_ERR != err) {
180  error("INPUT_TASK_FLAG_ERR: %i\n", err);
181  }
182  }
183  }
184  }
185 
186  // change systemState when Run-Bit = 1
187  if (ctrlReg & CTRL_REG_RS_MSK) {
188  modeBits = ((ctrlReg & CTRL_REG_MODE_MSK)>>2);
189  if ((modeBits & MODE_STOP_CON_RUN_MSK) == MODE_STOP) {
190  systemState.activeUseCase = STOP;
191  } else if ((modeBits & MODE_STOP_CON_RUN_MSK) == MODE_CON_RUN) {
192  systemState.activeUseCase = CONTINOUS;
193  } else if (modeBits == MODE_CH_OF_ST_1_4) {
194  systemState.activeUseCase = QUARTER_ROTATION;
195  } else if (modeBits == MODE_CH_OF_ST_1_2) {
196  systemState.activeUseCase = HALF_ROTATION;
197  } else if (modeBits == MODE_CH_OF_ST_1) {
198  systemState.activeUseCase = FULL_ROTATION;
199  } else if (modeBits == MODE_CH_OF_ST_2) {
200  systemState.activeUseCase = DOUBLE_ROTATION;
201  }
202  }
203  // new user input or stepsReg updated when motor running
204  if (newInput | ((ctrlReg & CTRL_REG_RS_MSK) & (oldStepsReg != stepsReg))){
205  // update content of mailbox to outputTask
206  outputTaskMailboxContent.ctrlReg = ctrlReg;
207  outputTaskMailboxContent.speedReg = speedReg;
208  // wait 1 second when steps reg updated
209  if((ctrlReg & CTRL_REG_RS_MSK) & (oldStepsReg != stepsReg)){
210  oldStepsReg = stepsReg;
211  outputTaskMailboxContent.stepsReg = stepsReg;
212  OSTimeDlyHMSM(0, 0, 1, 0);
213  }
214  outputTaskMailboxContent.systemState = systemState;
215  err = OSMboxPost(outputTaskMailbox, &outputTaskMailboxContent);
216  if (OS_NO_ERR != err) {
217  error("INPUT_TASK_MBOX_ERR: %i\n", err);
218  }
219  // write values of register copies into real registers
220  ctrlRegBitClr(CTRL_REG_0_6_MSK);
221  ctrlRegBitSet(ctrlReg & CTRL_REG_0_6_MSK);
222  speedRegSet(speedReg);
223  newInput = false;
224  }
225  }
226  }
227  }
