34 --! @brief Milestone 3
35 ENTITY milestone3 is
36  GENERIC
37  (
38  --! @brief Prescaler for PWM-signal.
39  --! @details For this purpose 2,5 ms are used as minimal pulse-width.
40  --! @details The prescaler is calculated with the given and desired frequency
41  --! via the following formula:
42  --! prescaler = f_clock [Hz] / f_prescaler [Hz]
43  --! e.g.: f_prescaler = 1/5 ms = 400 Hz
44  --! prescaler = 50 Mhz / 400 Hz = 125000
45  divider : integer := 125000
46  );
47  PORT
48  (
49  CLOCK_50_B5B : IN STD_LOGIC; --! component clock
50  CPU_RESET_n : IN STD_LOGIC; --! resets the component
51  SW : IN STD_LOGIC_VECTOR(9 DOWNTO 0); --! switch input
52  KEY : IN STD_LOGIC_VECTOR(3 DOWNTO 0); --! key input
53  HSMC_RX_P : OUT STD_LOGIC_VECTOR(16 DOWNTO 0); --! Motor_pwm1( bit 0 ) and Motor_pwm2( bit 1 )
54  HSMC_RX_N : OUT STD_LOGIC_VECTOR(16 DOWNTO 0); --! Motor_pwm3( bit 0 ) and Motor_pwm4( bit 1 )
55  HSMC_TX_N : OUT STD_LOGIC_VECTOR(16 DOWNTO 0); --! Motor_en_a( bit 2 )
56  HSMC_TX_P : OUT STD_LOGIC_VECTOR(16 DOWNTO 0); --! Motor_en_b( bit 3 )
57  LEDG : OUT STD_LOGIC_VECTOR(7 DOWNTO 0); --! green leds
58  LEDR : OUT STD_LOGIC_VECTOR(7 DOWNTO 0) --! red leds
59  );
60 END milestone3;
