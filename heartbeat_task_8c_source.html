27 void HeartbeatTask(void *pdata) {
28  uint8_t err;
29  OS_FLAGS heartbeatFlag;
30  heartbeatState_t heartbeatState = FIRST;
31  heartbeatState_t *heartbeatStatePtr = &heartbeatState;
32 
33  while (1) {
34  heartbeatFlag = OSFlagPend(heartbeatTaskFlagsGrp,
35  DEBUG_ON_EVENT,
36  OS_FLAG_WAIT_SET_ANY + OS_FLAG_CONSUME, 10, &err);
37  if (OS_NO_ERR != err && OS_TIMEOUT != err) {
38  error("There was an error while waiting for the event!");
39  } else {
40  if (heartbeatFlag & DEBUG_ON_EVENT)
41  debugAndHeartbeat(heartbeatStatePtr);
42  nextHeartbeatStep(heartbeatStatePtr);
43  }
44  }
45 }

47 void nextHeartbeatStep(heartbeatState_t *heartbeatStatePtr) {
48  switch (*heartbeatStatePtr) {
49  case FIRST:
50  PIO_HEX3_Set(LINE);
51  PIO_LED9_Set(0x1);
52  *heartbeatStatePtr = SECOND;
53  break;
54  case SECOND:
55  PIO_HEX3_Set(LOWER_O);
56  PIO_LED9_Set(0x0);
57  *heartbeatStatePtr = THIRD;
58  break;
59  case THIRD:
60  PIO_HEX3_Set(LINE);
61  PIO_LED9_Set(0x1);
62  *heartbeatStatePtr = FOURTH;
63  break;
64  case FOURTH:
65  PIO_HEX3_Set(UPPER_O);
66  PIO_LED9_Set(0x0);
67  *heartbeatStatePtr = FIRST;
68  break;
69  default:
70  break;
71  }
72  OSTimeDlyHMSM(0, 0, 1, 0);

75 void debugAndHeartbeat(heartbeatState_t *heartbeatStatePtr) {
76  uint32_t ctrlReg = 0;
77  uint8_t stepCounter = 0;
78 
79  for(stepCounter < 4) {
80  ctrlReg = ctrlRegGet();
81  // Is Run-Bit == 1? (RUN)
82  if (ctrlReg & CTRL_REG_RS_MSK) {
83  // Is Dir-Bit == 1? (RIGHT)
84  if (ctrlReg & CTRL_REG_LR_MSK) {
85  stepsRegSet(stepsRegGet() - 10);
86  }
87  stepsRegSet(stepsRegGet() + 10);
88  }
89 
90  // Debug things done -> next step of heartbeat-cycle...
91  ++stepCounter;
92  nextHeartbeatStep(heartbeatStatePtr);
93  }
94  // Set IR
95  ctrlReg |= CTRL_REG_IR_MSK;
96  ctrlRegSet(ctrlReg);
97 }
