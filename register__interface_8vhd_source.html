8 --! @brief Register component
9 --! @details Provides the `ctrlReg`, `speedReg`, `stepsReg` registers.
10 --! The `ctrlReg` can be set and cleared bitwise with the
11 --! `ctrlSetReg` and `ctrlClrReg`.
12 --! @details Addresses for register-access:
13 --! | address | register |
14 --! |:-------:|:-----------|
15 --! | `000` | `ctrlReg` |
16 --! | `001` |`ctrlSetReg`|
17 --! | `010` |`ctrlClrReg`|
18 --! | `011` |`speedReg` |
19 --! | `100` |`stepsReg` |

72 --! @brief Architecture of register_interface
73 --! @details realized components:
74 --! - all five registers
75 --! - 3 green LEDs showing speedReg with 0 .. 2
76 --! - 8 red LEDs showing ctrlReg
77 --! - generating motor interrupt
78 ARCHITECTURE my_register_interface OF register_interface IS
79 
80  --! internal signal representing the registers
81  SIGNAL ctrlReg : STD_LOGIC_VECTOR(7 DOWNTO 0);
82  SIGNAL speedReg : STD_LOGIC_VECTOR(2 DOWNTO 0);
83  SIGNAL stepsReg : STD_LOGIC_VECTOR(31 DOWNTO 0);
84 
85 BEGIN
86  --! @brief processing the tasks:
87  --! - interface to NIOS-processor
88  --! - writing registers
89  --! - reading registers
90  --! - implementing set and clear functionality
91  PROCESS(clock, reset_n, ce_n, read_n, write_n, addr, ctrlReg, ir, speedReg, stepsReg)
92  BEGIN
93  -- ctrlReg Register Write
94  IF (reset_n = '0') THEN
95  ctrlReg <= (others => '0');
96  ELSIF (rising_edge(clock)) THEN
97  ctrlReg <= ctrlReg;
98 
99  IF(ir = '1') THEN -- set IR-bit and reset R/S when mcu requests interrupt
100  ctrlReg(7) <= '1';
101  ctrlReg(0) <= '0';
102  END IF;
103 
104  IF (addr = B"000" AND write_n = '0' AND ce_n = '0') THEN
105  ctrlReg <= write_data(7 DOWNTO 0); -- overwrite complete ctrlReg
106  ELSIF(addr = B"001" AND write_n = '0' AND ce_n = '0') THEN
107  ctrlReg <= ctrlReg or write_data(7 DOWNTO 0); -- set ctrlReg bitwise
108  ELSIF(addr = B"010" AND write_n = '0' AND ce_n = '0') THEN
109  ctrlReg <= ctrlReg and (not write_data(7 DOWNTO 0)); -- clr ctrlReg
110  END IF;
111  END IF;
112 
113  -- speedReg Register Write
114  IF (reset_n = '0') THEN
115  speedReg <= (others => '0');
116  ELSIF (rising_edge(clock)) THEN
117  IF (addr = B"011" AND write_n = '0' AND ce_n = '0') THEN
118  speedReg <= write_data(2 DOWNTO 0);
119  else
120  speedReg <= speedReg;
121  END IF;
122  END IF;
123 
124  -- stepsReg Register Write
125  IF (reset_n = '0') THEN
126  stepsReg <= (others => '0');
127  ELSIF (rising_edge(clock)) THEN
128  stepsReg <= stepsReg;
129  IF(CONV_INTEGER(steps) /= 0) THEN -- write stepsReg from mcu when mcu input is not '0'
130  stepsReg <= steps;
131  END IF;
132  IF (addr = B"100" AND write_n = '0' AND ce_n = '0') THEN
133  stepsReg <= write_data(31 DOWNTO 0);
134  END IF;
135  END IF;
136 
137  -- Processor reads from Register
138  read_data <= (others => '0'); -- unused bits to 0
139  IF (read_n = '0' AND ce_n = '0') THEN
140  CASE addr is
141  WHEN B"000" =>
142  read_data(7 DOWNTO 0) <= ctrlReg;
143  WHEN B"011" =>
144  read_data(2 DOWNTO 0) <= speedReg;
145  WHEN B"100" =>
146  read_data(31 DOWNTO 0) <= stepsReg;
147  WHEN others =>
148  read_data(31 DOWNTO 0) <= (others => '0');
149  END CASE;
150  END IF;
151 
152  END process;
153 
154  -- set interrupt
155  irq <= ctrlReg(6) and ctrlReg(7);
156  -- clone ctrlReg to red leds
157  redleds <= ctrlReg;
158  -- clone speedReg to green leds
159  greenleds(2 DOWNTO 0) <= speedReg;
160  greenleds(7 DOWNTO 3) <= (others => '0');
161  -- extra mcu outputs
162  run <= ctrlReg(0);
163  direction <= ctrlReg(1);
164  mode <= ctrlReg(5 DOWNTO 2);
165  speed <= speedReg;
166 
167 END my_register_interface;
