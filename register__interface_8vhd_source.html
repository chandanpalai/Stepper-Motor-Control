8 --! @brief Register component
9 --! @details Provides the `ctrlReg`, `speedReg`, `stepsReg` registers.
10 --! The `ctrlReg` can be set and cleared bitwise with the
11 --! `ctrlSetReg` and `ctrlClrReg`.
12 --!
13 --! @details Combinations for register-access-mode:
14 --! |`ce_n`|`read_n`|`write_n`| Access |
15 --! |:----:|:------:|:-------:|:-------------|
16 --! | `0` | `0` | `0` | *none* |
17 --! | `0` | `0` | `1` | *none* |
18 --! | `0` | `1` | `0` | *none* |
19 --! | `0` | `1` | `1` | *none* |
20 --! | `1` | `0` | `0` | undefined |
21 --! | **1**| **0** | **1** | **readable** |
22 --! | **1**| **1** | **0** | **writeable**|
23 --! | `1` | `1` | `1` | undefined |
24 --!
25 --! @details Addesses for register-access:
26 --! | address | register |
27 --! |:-------:|:-----------|
28 --! | `000` | `ctrlReg` |
29 --! | `001` |`ctrlSetReg`|
30 --! | `010` |`ctrlClrReg`|
31 --! | `011` |`speedReg` |
32 --! | `100` |`stepsReg` |

75 --! @brief Architecture of register_interface
76 --! @details realized components:
77 --! - all five registers
78 --! - 3 green LEDs showing speedReg with 0 .. 2
79 --! - 8 red LEDs showing ctrlReg
80 --! - generating motor interrupt
81 architecture my_register_interface of register_interface is
82 
83  --! internal signal representing the registers
84  signal ctrlReg : std_logic_vector(7 downto 0);
85  signal ctrlSetReg : std_logic_vector(7 downto 0);
86  signal ctrlClrReg : std_logic_vector(7 downto 0);
87  signal speedReg : std_logic_vector(2 downto 0);
88  signal stepsReg : std_logic_vector(31 downto 0);
89 
90 
91 begin
92  --! @brief processing the tasks:
93  --! - interface to NIOS-processor
94  --! - writing registers
95  --! - reading registers
96  --! - implementing set and clear functionality
97  process(clock, reset_n, ce_n, read_n, write_n, addr, ctrlReg, ctrlSetReg,
98  ctrlClrReg, speedReg, stepsReg)
99  begin
100  if (reset_n = '0') then
101  ctrlReg(7 downto 0) <= (others => '0');
102  ctrlSetReg(7 downto 0) <= (others => '0');
103  ctrlClrReg(7 downto 0) <= (others => '0');
104  speedReg(2 downto 0) <= (others => '0');
105  stepsReg(31 downto 0) <= (others => '0');
106  read_data(31 downto 0) <= (others => '0');
107  elsif (rising_edge(clock)) then
108  -- Processor writes to Register
109  if (write_n = '0' AND read_n = '1' AND ce_n = '0') then
110  case addr is
111  when B"000" =>
112  ctrlReg <= write_data(7 downto 0);
113  when B"001" =>
114  ctrlSetReg <= write_data(7 downto 0);
115  when B"010" =>
116  ctrlClrReg <= write_data(7 downto 0);
117  when B"011" =>
118  speedReg <= write_data(2 downto 0);
119  when B"100" =>
120  stepsReg <= write_data(31 downto 0);
121  when others =>
122  end case;
123  end if;
124 
125  -- set and clear functionality for ctrlReg
126  if (CONV_INTEGER(ctrlSetReg) /= 0) then
127  ctrlReg <= ctrlReg or ctrlSetReg;
128  ctrlSetReg <= (others => '0');
129  end if;
130 
131  if (CONV_INTEGER(ctrlClrReg) /= 0) then
132  ctrlReg <= ctrlReg and (not ctrlClrReg);
133  ctrlClrReg <= (others => '0');
134  end if;
135 
136  end if;
137 
138  -- Processor reads from Register
139  if (write_n = '1' AND read_n = '0' AND ce_n = '0') then
140  read_data <= (others => '0'); -- unused bits to 0
141  case addr is
142  when B"000" =>
143  read_data(7 downto 0) <= ctrlReg;
144  when B"001" =>
145  read_data(7 downto 0) <= ctrlSetReg;
146  when B"010" =>
147  read_data(7 downto 0) <= ctrlClrReg;
148  when B"011" =>
149  read_data(2 downto 0) <= speedReg;
150  when B"100" =>
151  read_data(31 downto 0) <= stepsReg;
152  when others =>
153  read_data(31 downto 0) <= (others => '0');
154  end case;
155  end if;
156 
157  end process;
158 
159  -- set interrupt
160  irq <= ctrlReg(6) and ctrlReg(7);
161  -- clone ctrlReg to red leds
162  redleds <= ctrlReg;
163  -- clone speedReg to green leds
164  greenleds(2 downto 0) <= speedReg;
165  greenleds(7 downto 3) <= (others => '0');
166 
167 end my_register_interface;
